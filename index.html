<script>

const months = [

    {name:"Septiembre 2025", days:30},

    {name:"Octubre 2025", days:31},

    {name:"Noviembre 2025", days:30},

    {name:"Diciembre 2025", days:31},

    {name:"Enero 2026", days:31},

    {name:"Febrero 2026", days:28},

    {name:"Marzo 2026", days:31},

    {name:"Abril 2026", days:30}

];


let currentMonthIndex = 0;


function saveData() {

    const allCells = document.querySelectorAll('.mini-cell');

    const data = {};

    allCells.forEach(cell => {

        const key = cell.dataset.key;

        data[key] = cell.innerText;

    });

    localStorage.setItem('calendarData', JSON.stringify(data));

    updateCellColors();

}


function loadData() {

    const data = JSON.parse(localStorage.getItem('calendarData') || '{}');

    const allCells = document.querySelectorAll('.mini-cell');

    allCells.forEach(cell => {

        const key = cell.dataset.key;

        if(data[key]) cell.innerText = data[key];

    });

    updateCellColors();

}


function updateCellColors() {

    const allCells = document.querySelectorAll('.mini-cell');

    allCells.forEach(cell => {

        cell.classList.remove('filled-instructor','filled-sub');

        if(cell.innerText.trim() !== '') {

            if(cell.dataset.role === 'instructor') cell.classList.add('filled-instructor');

            else if(cell.dataset.role === 'sub') cell.classList.add('filled-sub');

        }

    });

}


function clearData() {

    if(confirm('¿Deseas borrar todos los datos del calendario?')) {

        localStorage.removeItem('calendarData');

        renderMonth();

    }

}


function renderMonth() {

    const month = months[currentMonthIndex];

    const container = document.getElementById('monthContainer');

    container.innerHTML = `<div class="month-header">${month.name}</div>`;

    const calendarDiv = document.createElement('div');

    calendarDiv.className = 'calendar';

    ["Lunes","Martes","Miércoles","Jueves","Viernes"].forEach(day => {

        const header = document.createElement('div');

        header.className = 'day-header';

        header.innerText = day;

        calendarDiv.appendChild(header);

    });


    for(let day=1; day<=month.days; day++){

        const dayCell = document.createElement('div');

        dayCell.className = 'day-cell';

        dayCell.innerHTML = `<div class="day-number">${day}</div>`;

        const miniCells = document.createElement('div');

        miniCells.className = 'mini-cells';

        for(let i=1;i<=4;i++){

            const mini = document.createElement('div');

            mini.className = 'mini-cell';

            mini.contentEditable = "true";

            mini.dataset.key = `${month.name}-day${day}-cell${i}`;

            mini.dataset.role = i<=2 ? 'instructor' : 'sub';

            mini.innerHTML = i<=2 ? `<div class="mini-cell-label">Instructor</div>` : `<div class="mini-cell-label">Sub</div>`;

            mini.addEventListener('input', saveData);

            miniCells.appendChild(mini);

        }

        dayCell.appendChild(miniCells);

        calendarDiv.appendChild(dayCell);

    }

    container.appendChild(calendarDiv);

    loadData();

}


document.getElementById('prevMonth').addEventListener('click',()=>{

    if(currentMonthIndex>0) currentMonthIndex--;

    renderMonth();

});

document.getElementById('nextMonth').addEventListener('click',()=>{

    if(currentMonthIndex<months.length-1) currentMonthIndex++;

    renderMonth();

});

document.getElementById('clearData').addEventListener('click',clearData);


renderMonth();

</script>
